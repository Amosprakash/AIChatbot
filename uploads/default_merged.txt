### Start of Document: Kafka Queue in C#.txt ###
Implement Kafka queue in Message Queue

Steps In System:
	1.Setup Apache Kafka server in the System  (Need Java)
		
			link to download:https://kafka.apache.org/downloads.

	2.Create kafka server , Topic Name , partition and Replication for the Kafka Queue in cmd.
		
		a.kafka-topics.bat --create --zookeeper localhost:2181 --replication-factor 1 --partitions 1 --topic test

	3.Add the server name ,Topic name and Consuemer Group Id in web.config.
Steps in Visual Studio:

	1.Create Producer.cs and Consumer.cs in the previously Created MSMQ Class in ASync and Sync method 

		a. Install-Package Confluent.Kafka in Nuget Package console

	2.Change All the references that made for Msmq in the project using the kafka queue. 

	3.In addition to integrating Kafka queue Can be  create this Producer and Consumer Function as  a service.

Difference from the MSMQ:

	1.It doesn't create a message queue system in the task service in computer Management.
	
CMD :
	1.we can reterive the Message status using CMD 

		 a.List Topics: kafka-topics.sh --list --bootstrap-server localhost:9092
		 b. Describe a Topic:kafka-topics.sh --describe --topic <topic_name> --bootstrap-server localhost:9092
		 c.Consume Messages from a Topic:kafka-console-consumer.sh --topic <topic_name> --bootstrap-server localhost:9092 --from-beginning
		 d. Produce Messages to a Topic:kafka-console-producer.sh --topic <topic_name> --bootstrap-server localhost:9092
		 e.Get Topic Offsets:kafka-run-class.sh kafka.tools.GetOffsetShell --broker-list localhost:9092 --topic <topic_name> --time -1
		 f.Delete a Topic:kafka-topics.sh --delete --topic <topic_name> --bootstrap-server localhost:9092
	
	Reference : https://www.codemag.com/Article/2201061/Working-with-Apache-Kafka-in-ASP.NET-6-Core

from pyhtml2pdf import converter
from urllib.parse import urlparse, parse_qs

app = Flask(__name__)

# Define the two folder paths for saving PDFs
FOLDER_SPINCO = 'spinco_folder'
FOLDER_OTHER = 'other_folder'

# Check if folders exist and create them if not
if not os.path.exists(FOLDER_SPINCO):
    os.makedirs(FOLDER_SPINCO)

import os
import time
from flask import Flask, render_template, jsonify, request, send_file
from pyhtml2pdf import converter
from urllib.parse import urlparse, parse_qs

app = Flask(__name__)

# Define the two folder paths for saving PDFs
FOLDER_SPINCO = 'spinco_folder'
FOLDER_OTHER = 'other_folder'

# Check if folders exist and create them if not
if not os.path.exists(FOLDER_SPINCO):
    os.makedirs(FOLDER_SPINCO)
if not os.path.exists(FOLDER_OTHER):
    os.makedirs(FOLDER_OTHER)

def get_unique_filename(folder, filename):
    """
    Checks if the file already exists in the folder. If it does, appends a number (1, 2, 3, ...)
    to the filename until a unique name is found.
    """
    base_name, ext = os.path.splitext(filename)
    counter = 1
    new_filename = filename
    while os.path.exists(os.path.join(folder, new_filename)):
        new_filename = f"{base_name}({counter}){ext}"
        counter += 1
    return new_filename

@app.route("/")
def index():
    return "url_to_pdf_convertor"

@app.route("/url")
def change_ip():
    # Get the 'name' parameter from the URL (which contains the URL to be converted)
    name = request.args.get('name')

    # Check if the URL contains "SPINCOTECHLLP"
    if "SPINCOTECHLLP" in name:
        folder = FOLDER_SPINCO  # Save to spinco_folder
    else:
        folder = FOLDER_OTHER  # Save to other_folder

    # Generate the PDF file name based on the query parameters
    if "qNo" in name:
        url = name
        parsed_url = urlparse(url)
        # Extract the qNo value from the query string and clean it up (e.g., replace slashes with dashes)
        hello_world = parse_qs(parsed_url.query)['qNo'][0].replace("/", "-")
        output_pdf = hello_world + ".pdf"
    else:
        output_pdf = 'sample.pdf'

    # Convert the URL to a PDF
    converter.convert(name, output_pdf, timeout=4, print_options={"scale": 1, "paperHeight": 12})

    # Get a unique filename if a file with the same name exists
    output_pdf_unique = get_unique_filename(folder, output_pdf)

    # Define the output file path based on the chosen folder
    output_pdf_path = os.path.join(folder, output_pdf_unique)

    # Move the generated PDF to the correct folder
    os.rename(output_pdf, output_pdf_path)

    # Send the PDF file as a download
    return send_file(output_pdf_path, as_attachment=True)

if __name__ == "__main__":
    app.run(host='0.0.0.0',port='3000')

import os
import time
from flask import Flask, render_template, jsonify, request, send_file
from pyhtml2pdf import converter
from urllib.parse import urlparse, parse_qs

app = Flask(__name__)

# Define the two folder paths for saving PDFs
FOLDER_SPINCO = 'spinco_folder'
FOLDER_OTHER = 'other_folder'

# Check if folders exist and create them if not
if not os.path.exists(FOLDER_SPINCO):
    os.makedirs(FOLDER_SPINCO)
if not os.path.exists(FOLDER_OTHER):
    os.makedirs(FOLDER_OTHER)

def get_unique_filename(folder, filename):
    """
    Checks if the file already exists in the folder. If it does, appends a number (1, 2, 3, ...)
    to the filename until a unique name is found.
    """
    base_name, ext = os.path.splitext(filename)
    counter = 1
    new_filename = filename
    while os.path.exists(os.path.join(folder, new_filename)):
        new_filename = f"{base_name}({counter}){ext}"
        counter += 1
    return new_filename

@app.route("/")
def index():
    return "url_to_pdf_convertor"

@app.route("/url")
def change_ip():
    # Get the 'name' parameter from the URL (which contains the URL to be converted)
    name = request.args.get('name')

    # Check if the URL contains "SPINCOTECHLLP"
    if "SPINCOTECHLLP" in name:
        folder = FOLDER_SPINCO  # Save to spinco_folder
    else:
        folder = FOLDER_OTHER  # Save to other_folder

    # Generate the PDF file name based on the query parameters
    if "qNo" in name:
        url = name
        parsed_url = urlparse(url)
        # Extract the qNo value from the query string and clean it up (e.g., replace slashes with dashes)
        hello_world = parse_qs(parsed_url.query)['qNo'][0].replace("/", "-")
        output_pdf = hello_world + ".pdf"
    else:
        output_pdf = 'sample.pdf'

    # Convert the URL to a PDF
    converter.convert(name, output_pdf, timeout=4, print_options={"scale": 1, "paperHeight": 12})

    # Get a unique filename if a file with the same name exists
    output_pdf_unique = get_unique_filename(folder, output_pdf)

    # Define the output file path based on the chosen folder
    output_pdf_path = os.path.join(folder, output_pdf_unique)

    # Move the generated PDF to the correct folder
    os.rename(output_pdf, output_pdf_path)

    # Send the PDF file as a download
    return send_file(output_pdf_path, as_attachment=True)

if __name__ == "__main__":
    app.run(host='0.0.0.0',port='3000')
